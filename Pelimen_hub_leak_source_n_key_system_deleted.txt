local function createTab(name, icon, index)
    local btn = Instance.new("TextButton", tabContainer)
    btn.Size = UDim2.new(1, -10, 0, 45)
    btn.Position = UDim2.new(0, 5, 0, 5 + (index * 50))
    btn.Text = icon .. " " .. name
    btn.BackgroundColor3 = Color3.fromRGB(30, 30, 50)
    btn.TextColor3 = Color3.fromRGB(150, 150, 150)
    btn.Font = Enum.Font.GothamBold
    btn.TextSize = 14
    btn.BorderSizePixel = 0
    btn.AutoButtonColor = false
    btn.TextXAlignment = Enum.TextXAlignment.Center
    btn.ZIndex = 102
    
    local btnCorner = Instance.new("UICorner", btn)
    btnCorner.CornerRadius = UDim.new(0, 10)
    
    tabButtons[name] = btn
    
    btn.MouseButton1Click:Connect(function()
        currentTab = name
        for tabName, tabBtn in pairs(tabButtons) do
            if tabName == name then
                tween(tabBtn, {BackgroundColor3 = Color3.fromRGB(255, 50, 120), TextColor3 = Color3.fromRGB(255, 255, 255)})
            else
                tween(tabBtn, {BackgroundColor3 = Color3.fromRGB(30, 30, 50), TextColor3 = Color3.fromRGB(150, 150, 150)})
            end
        end
        for _, child in pairs(contentFrame:GetChildren()) do
            if child:IsA("ScrollingFrame") then
                child.Visible = false
            end
        end
        local targetFrame = contentFrame:FindFirstChild(name .. "Content")
        if targetFrame then
            targetFrame.Visible = true
        end
    end)
    
    return btn
end

local function toggleMainFrame()
    if not scriptLoaded then return end
    
    if mainFrame.Visible then
        mainFrame:TweenSizeAndPosition(
            UDim2.new(0, 0, 0, 0),
            UDim2.new(0.5, 0, 0.5, 0),
            Enum.EasingDirection.Out,
            Enum.EasingStyle.Quad,
            0.3,
            true,
            function()
                mainFrame.Visible = false
                mainFrame.Size = UDim2.new(0, 550, 0, 400)
                mainFrame.Position = UDim2.new(0.5, -275, 0.5, -200)end
        )
    else
        mainFrame.Size = UDim2.new(0, 0, 0, 0)
        mainFrame.Position = UDim2.new(0.5, 0, 0.5, 0)
        mainFrame.Visible = true
        mainFrame:TweenSizeAndPosition(
            UDim2.new(0, 550, 0, 400),
            UDim2.new(0.5, -275, 0.5, -200),
            Enum.EasingDirection.Out,
            Enum.EasingStyle.Quad,
            0.3,
            true
        )
    end
end

local flying = false
local flyConnection

local function startFlying()
    flying = true
    local char = plr.Character
    if not char then return end
    
    local hrp = char:FindFirstChild("HumanoidRootPart")
    if not hrp then return end
    
    local bg = Instance.new("BodyGyro", hrp)
    bg.MaxTorque = Vector3.new(9e9, 9e9, 9e9)
    bg.P = 9e4
    
    local bv = Instance.new("BodyVelocity", hrp)
    bv.MaxForce = Vector3.new(9e9, 9e9, 9e9)
    bv.Velocity = Vector3.new(0, 0, 0)
    
    flyConnection = run.Heartbeat:Connect(function()
        if not flying or not SETTINGS.FlyMode then
            bg:Destroy()
            bv:Destroy()
            if flyConnection then flyConnection:Disconnect() end
            flying = false
            return
        end
        
        local cam = workspace.CurrentCamera
        bg.CFrame = cam.CFrame
        
        local speed = flySpeed * 50
        local velocity = Vector3.new(0, 0, 0)
        
        if uis:IsKeyDown(Enum.KeyCode.W) then
            velocity = velocity + (cam.CFrame.LookVector * speed)
        end
        if uis:IsKeyDown(Enum.KeyCode.S) then
            velocity = velocity - (cam.CFrame.LookVector * speed)
        end
        if uis:IsKeyDown(Enum.KeyCode.A) then
            velocity = velocity - (cam.CFrame.RightVector * speed)
        end
        if uis:IsKeyDown(Enum.KeyCode.D) then
            velocity = velocity + (cam.CFrame.RightVector * speed)
        end
        if uis:IsKeyDown(Enum.KeyCode.Space) then
            velocity = velocity + Vector3.new(0, speed, 0)
        end
        if uis:IsKeyDown(Enum.KeyCode.LeftShift) then
            velocity = velocity - Vector3.new(0, speed, 0)
        end
        
        bv.Velocity = velocity
    end)
end

    end
end)

local function blurGodText(obj)
    pcall(function()
        if obj:IsA("TextLabel") or obj:IsA("TextButton") or obj:IsA("TextBox") then
            if obj.Text:match("[Gg][Oo][Dd]") then
                obj.Text = string.gsub(obj.Text, "[Gg][Oo][Dd]", "***")
            end
        end
    end)
end

local function monitorDescendants(parent)
    pcall(function()
        for _, obj in pairs(parent:GetDescendants()) do
            blurGodText(obj)
        end
        parent.DescendantAdded:Connect(function(desc)
            task.wait(0.1)
            blurGodText(desc)
        end)
    end)
end
